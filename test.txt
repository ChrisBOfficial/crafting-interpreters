/*
    Test file for running the Lox interpreter
*/

class Food {
    eat() {
        print "It is certainly edible";
    }
}

class Cake <: Food {
    init(flavor) {
        this.flavor = flavor;
    }

    taste() {
        var adjective = "delicious";
        print "The " + this.flavor + " cake is " + adjective + "!";
        super.eat();
    }
}

var cake = Cake("Angel food");
cake.taste();
